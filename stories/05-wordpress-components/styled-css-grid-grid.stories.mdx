## Config
```json
{
  "align_content": {
    "description": "The align-content CSS property. Not provided by default.",
    "type": "string"
  },
  "areas": {
    "description": "The grid-template-areas CSS property. Pass an array of strings, e.g. [\"a a\", \"b c\"]. Not provided by default.",
    "type": "array"
  },
  "column_gap": {
    "description": "The column-gap CSS property. Not provided by default.",
    "type": "string"
  },
  "columns": {
    "description": "The grid-template-columns CSS property. When a number is passed it is a shorthand to specify the number of columns. Default is 12.",
    "type": "string"
  },
  "flow": {
    "description": "The grid-auto-flow CSS property. Default is \"row\".",
    "type": "string"
  },
  "gap": {
    "description": "The grid-gap CSS property. Default is \"8px\".",
    "type": "string"
  },
  "height": {
    "description": "The height CSS property. Default is \"auto\".",
    "type": "string"
  },
  "justify_content": {
    "description": "The justify-content CSS property. Not provided by default.",
    "type": "string"
  },
  "min_row_height": {
    "description": "Minimum height of each row. Default is \"20px\".",
    "type": "string"
  },
  "row_gap": {
    "description": "The row-gap CSS property. Not provided by default.",
    "type": "string"
  },
  "rows": {
    "description": "The grid-template-rows CSS property. When a number is passed it is a shorthand to specify the number of rows. Not provided by default.",
    "type": "string"
  },
  "style": {
    "default": [],
    "type": "array"
  }
}
```
